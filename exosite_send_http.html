Tutorial

This page explains how to use the Exosite Ready SDK through a a very simple example application
The following source is the source code of an application using the SDK. You will find detailed explanation below:

Source
#include <stdio.h>
#include <lib/type.h>
#include <lib/error.h>
#include <lib/debug.h>
#include <exosite_api.h>
static char counter_str[16];
static int counter;
static void on_read(int status, const char *alias, const char *value)
{
if (status == ERR_SUCCESS)
printf("Read \"%s\": %s\n", alias, value);
}
static void on_write(int status, const char *alias)
{
if (status == ERR_SUCCESS)
printf("Write \"%s\"\n", alias);
}
int main(int argc, char *argv[])
{
struct exosite_class *exo;
enum exosite_device_status status;
int error;
char *vendor, *model, *sn;
if (argc < 4) {
printf("Please specify the VENDOR, MODEL, SERIAL_NUMBER of your device and run again\n");
printf("Without VENDOR, MODEL, SERIAL_NUMBER the test cannot connect connect to the Exosite cloud\n");
return ERR_FAILURE;
}
vendor = argv[1];
model = argv[2];
sn = argv[3];
error = exosite_sdk_init(printf, NULL);
if (error)
return ERR_FAILURE;
info_log(DEBUG_APPLICATION, ("Send http example\n"));
error = exosite_new(&exo, vendor, model, sn, APP_PROTO_HTTP);
if (error)
return ERR_FAILURE;
do {
status = exosite_get_status(exo);
exosite_delay_and_poll(exo, 200);
} while (status != DEVICE_STATUS_ACTIVATED);
while (1) {
sprintf(counter_str, "%d", counter++);
exosite_write(exo, "count", counter_str, on_write);
exosite_delay_and_poll(exo, 2000);
exosite_read(exo, "count", on_read);
exosite_delay_and_poll(exo, 2000);
}
return ERR_SUCCESS;
}

Send data over HTTP

Create the callback function that will be called when the write operation is finished. It will get the status and the alias of the data source.

static void on_write(int status, const char *alias)
{
if (status == ERR_SUCCESS)
printf("Write \"%s\"\n", alias);
}

In the main function we have to declare a couple of variables first.

struct exosite_class *exo;
enum exosite_device_status status;
int error;

The next step is the SDK initialization.

error = exosite_sdk_init(printf, NULL);
if (error)
return ERR_FAILURE;

Having initialized the SDK we can create an instance

error = exosite_new(&exo, vendor, model, sn, APP_PROTO_HTTP);
if (error)
return ERR_FAILURE;

Wait until the activation is finished

do {
status = exosite_get_status(exo);
exosite_delay_and_poll(exo, 200);
} while (status != DEVICE_STATUS_ACTIVATED);

In an infinite loop, we write value into the "Counter" data source in every two seconds. For waiting, we use exosite_delay_and_poll function.

while (1) {
sprintf(counter_str, "%d", counter++);
exosite_write(exo, "count", counter_str, on_write);
exosite_delay_and_poll(exo, 2000);
}
